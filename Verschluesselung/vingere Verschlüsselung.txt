
vigenereVer nachricht schluessel  =  vigenere nachricht schluessel "" 1
vigenereEnt nachricht schluessel  = vigenere nachricht schluessel "" (-1)


vigenere  nachricht  [] rest richtung  = []
vigenere []  schluessel rest richtung = []
vigenere nachricht schluessel [] richtung= vigenere nachricht schluessel schluessel richtung
vigenere nachricht schluessel rest richtung= (toEnum(erkenneBereich (head nachricht) (head rest) (richtung))::Char) : vigenere (tail nachricht) schluessel (tail rest) richtung


erkenneBereich a b c | zahlA >= 65 && zahlA <= 90  || zahlA == 196 || zahlA == 214 || zahlA == 220   = berechneAsciiBuchstabe (mod (anzahlGroßer + c*(mod zahlB anzahlGroßer) + berechneAsciiZahl zahlA 0) anzahlGroßer) 0
                    | zahlA >= 97 && zahlA <= 122 || zahlA == 228 || zahlA == 246 || zahlA == 252 || zahlA == 223   = berechneAsciiBuchstabe (mod (anzahlKleiner + c*(mod zahlB anzahlKleiner) + berechneAsciiZahl zahlA 1) anzahlKleiner) 1
                    | otherwise = zahlA
    where zahlA = fromEnum a :: Int
          zahlB = fromEnum b :: Int
          anzahlKleiner = 30
          anzahlGroßer = 29



berechneAsciiZahl neuerWert  0 | neuerWert == 196 = 26
                             | neuerWert == 214 = 27
                             | neuerWert == 220 = 28
                             | otherwise = neuerWert -65

berechneAsciiZahl neuerWert  1 | neuerWert == 228 = 26
                             | neuerWert == 246 = 27
                             | neuerWert == 252 = 28
                             | neuerWert == 223 = 29
                             | otherwise = neuerWert -97


berechneAsciiBuchstabe neuerWert  0 | mod neuerWert 29 == 26 = 196
                                    | mod neuerWert 29 == 27 = 214
                                    | mod neuerWert 29 == 28  = 220
                                    | otherwise = 65 + mod neuerWert 29

berechneAsciiBuchstabe neuerWert  1 | mod neuerWert 30 == 26 = 228
                                    | mod neuerWert 30 == 27 = 246
                                    | mod neuerWert 30 == 28 = 252
                                    | mod neuerWert 30 == 29 = 223
                                    | otherwise = 97 + mod neuerWert 29
